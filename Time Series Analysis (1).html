import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.seasonal import seasonal_decompose

# Load dataset
df = pd.read_csv(r"D:/Excel/iris.csv")

# Create a time-like index
df['Index'] = pd.RangeIndex(start=0, stop=len(df), step=1)
df.set_index('Index', inplace=True)

# Select a numerical column for analysis
series = df['sepal_length']

# Plot Time Series

plt.figure(figsize=(10,5))
plt.plot(series)
plt.title("Time Series Plot of Sepal Length")
plt.xlabel("Index (Time)")
plt.ylabel("Sepal Length")
plt.show()


# Decompose the Series

decomposition = seasonal_decompose(series, model='additive', period=10)

decomposition.plot()
plt.show()


# Moving Average Smoothing

df['Moving_Average'] = series.rolling(window=10).mean()

plt.figure(figsize=(10,5))
plt.plot(series, label='Original Data')
plt.plot(df['Moving_Average'], label='Moving Average')
plt.title("Moving Average Smoothing")
plt.xlabel("Index (Time)")
plt.ylabel("Sepal Length")
plt.legend()
plt.show()

import pandas as pd
import matplotlib.pyplot as plt
from statsmodels.tsa.seasonal import seasonal_decompose

data= pd.read_csv(r"D:/Excel/3) Sentiment dataset.csv")

data.columns = data.columns.str.strip()
data['Timestamp'] = pd.to_datetime(data['Timestamp'], errors='coerce')
data.set_index('Timestamp', inplace=True)


data = data.sort_index()

# Time series plot
plt.figure()
plt.plot(data['Likes'])
plt.title("Likes Over Time")
plt.xlabel("Time")
plt.ylabel("Likes")
plt.show()

# Moving average
data['Likes_MA'] = data['Likes'].rolling(window=7).mean()

plt.figure()
plt.plot(data['Likes'], label='Original')
plt.plot(data['Likes_MA'], label='7-day Moving Average')
plt.legend()
plt.title("Smoothed Likes Trend")
plt.show()

# Decomposition
likes_series = data['Likes'].dropna()

decomposition = seasonal_decompose(
    likes_series,
    model='additive',
    period=30
)

decomposition.plot()
plt.show()
